
IF NOT EXISTS (SELECT name FROM sys.databases WHERE name = 'OSKPMDbProduction')
BEGIN
    CREATE DATABASE OSKPMDbProduction;
END


USE OSKPMDbProduction;


CREATE TABLE [AspNetRoles] (
    [Id]               uniqueidentifier NOT NULL,
    [Name]             nvarchar(256)    NULL,
    [NormalizedName]   nvarchar(256)    NULL,
    [ConcurrencyStamp] nvarchar(max)    NULL,
    CONSTRAINT [PK_AspNetRoles] PRIMARY KEY ([Id])
);


CREATE TABLE [AspNetUsers] (
    [Id]                   uniqueidentifier NOT NULL,
    [FirstName]            nvarchar(30)     NOT NULL,
    [LastName]             nvarchar(30)     NULL,
    [PhNumber]             nvarchar(30)     NULL,
    [PinHash]              nvarchar(512)    NOT NULL,
    [CreatedAt]            datetime2        NOT NULL,
    [LastActive]           datetime2        NULL,
    [UserName]             nvarchar(256)    NULL,
    [NormalizedUserName]   nvarchar(256)    NULL,
    [Email]                nvarchar(256)    NULL,
    [NormalizedEmail]      nvarchar(256)    NULL,
    [EmailConfirmed]       bit              NOT NULL,
    [PasswordHash]         nvarchar(max)    NULL,
    [SecurityStamp]        nvarchar(max)    NULL,
    [ConcurrencyStamp]     nvarchar(max)    NULL,
    [PhoneNumber]          nvarchar(max)    NULL,
    [PhoneNumberConfirmed] bit              NOT NULL,
    [TwoFactorEnabled]     bit              NOT NULL,
    [LockoutEnd]           datetimeoffset   NULL,
    [LockoutEnabled]       bit              NOT NULL,
    [AccessFailedCount]    int              NOT NULL,
    CONSTRAINT [PK_AspNetUsers] PRIMARY KEY ([Id])
);


CREATE TABLE [AspNetRoleClaims] (
    [Id]         int              NOT NULL IDENTITY,
    [RoleId]     uniqueidentifier NOT NULL,
    [ClaimType]  nvarchar(max)    NULL,
    [ClaimValue] nvarchar(max)    NULL,
    CONSTRAINT [PK_AspNetRoleClaims] PRIMARY KEY ([Id]),
    CONSTRAINT [FK_AspNetRoleClaims_AspNetRoles_RoleId]
        FOREIGN KEY ([RoleId]) REFERENCES [AspNetRoles] ([Id]) ON DELETE CASCADE
);


CREATE TABLE [AspNetUserClaims] (
    [Id]         int              NOT NULL IDENTITY,
    [UserId]     uniqueidentifier NOT NULL,
    [ClaimType]  nvarchar(max)    NULL,
    [ClaimValue] nvarchar(max)    NULL,
    CONSTRAINT [PK_AspNetUserClaims] PRIMARY KEY ([Id]),
    CONSTRAINT [FK_AspNetUserClaims_AspNetUsers_UserId]
        FOREIGN KEY ([UserId]) REFERENCES [AspNetUsers] ([Id]) ON DELETE CASCADE
);


CREATE TABLE [AspNetUserLogins] (
    [LoginProvider]       nvarchar(128)    NOT NULL,
    [ProviderKey]         nvarchar(128)    NOT NULL,
    [ProviderDisplayName] nvarchar(max)    NULL,
    [UserId]              uniqueidentifier NOT NULL,
    CONSTRAINT [PK_AspNetUserLogins] PRIMARY KEY ([LoginProvider], [ProviderKey]),
    CONSTRAINT [FK_AspNetUserLogins_AspNetUsers_UserId]
        FOREIGN KEY ([UserId]) REFERENCES [AspNetUsers] ([Id]) ON DELETE CASCADE
);



CREATE TABLE [AspNetUserRoles] (
    [UserId] uniqueidentifier NOT NULL,
    [RoleId] uniqueidentifier NOT NULL,
    CONSTRAINT [PK_AspNetUserRoles] PRIMARY KEY ([UserId], [RoleId]),
    CONSTRAINT [FK_AspNetUserRoles_AspNetRoles_RoleId]
        FOREIGN KEY ([RoleId]) REFERENCES [AspNetRoles] ([Id]) ON DELETE CASCADE,
    CONSTRAINT [FK_AspNetUserRoles_AspNetUsers_UserId]
        FOREIGN KEY ([UserId]) REFERENCES [AspNetUsers] ([Id]) ON DELETE CASCADE
);


CREATE TABLE [AspNetUserTokens] (
    [UserId]        uniqueidentifier NOT NULL,
    [LoginProvider] nvarchar(128)    NOT NULL,
    [Name]          nvarchar(128)    NOT NULL,
    [Value]         nvarchar(max)    NULL,
    CONSTRAINT [PK_AspNetUserTokens] PRIMARY KEY ([UserId], [LoginProvider], [Name]),
    CONSTRAINT [FK_AspNetUserTokens_AspNetUsers_UserId]
        FOREIGN KEY ([UserId]) REFERENCES [AspNetUsers] ([Id]) ON DELETE CASCADE
);


CREATE TABLE [Credentials] (
    [CredentialId]   uniqueidentifier NOT NULL,
    [CredentialName] nvarchar(50)     NOT NULL,
    [UserId]         uniqueidentifier NOT NULL,
    [CreatedAt]      datetime2        NOT NULL,
    [LastEditedAt]   datetime2        NOT NULL,
    CONSTRAINT [PK_Credentials] PRIMARY KEY ([CredentialId]),
    CONSTRAINT [FK_Credentials_AspNetUsers_UserId]
        FOREIGN KEY ([UserId]) REFERENCES [AspNetUsers] ([Id]) ON DELETE CASCADE
);


CREATE TABLE [CredentialAccesses] (
    [CredentialAccessId] uniqueidentifier NOT NULL,
    [CredentialId]       uniqueidentifier NOT NULL,
    [UserId]             uniqueidentifier NOT NULL,
    [PermissionLevel]    nvarchar(50)     NOT NULL,
    [ExpireAt]           datetime2        NOT NULL,
    [SharedAt]           datetime2        NOT NULL,
    [SharedByUserId]     uniqueidentifier NOT NULL,
    CONSTRAINT [PK_CredentialAccesses] PRIMARY KEY ([CredentialAccessId]),
    CONSTRAINT [FK_CredentialAccesses_AspNetUsers_SharedByUserId]
        FOREIGN KEY ([SharedByUserId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_CredentialAccesses_AspNetUsers_UserId]
        FOREIGN KEY ([UserId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_CredentialAccesses_Credentials_CredentialId]
        FOREIGN KEY ([CredentialId]) REFERENCES [Credentials] ([CredentialId]) ON DELETE CASCADE
);


CREATE TABLE [WebCredentials] (
    [WebCredentialId] uniqueidentifier NOT NULL,
    [CredentialId]    uniqueidentifier NOT NULL,
    [Url]             nvarchar(200)    NOT NULL,
    [Username]        nvarchar(100)    NOT NULL,
    [PasswordHash]    nvarchar(512)    NOT NULL,
    [Notes]           nvarchar(100)    NULL,
    [CreatedAt]       datetime2        NOT NULL,
    [CreatorId]       uniqueidentifier NOT NULL,
    [EditedAt]        datetime2        NULL,
    [EditorId]        uniqueidentifier NULL,
    CONSTRAINT [PK_WebCredentials] PRIMARY KEY ([WebCredentialId]),
    CONSTRAINT [FK_WebCredentials_AspNetUsers_CreatorId]
        FOREIGN KEY ([CreatorId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_WebCredentials_AspNetUsers_EditorId]
        FOREIGN KEY ([EditorId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_WebCredentials_Credentials_CredentialId]
        FOREIGN KEY ([CredentialId]) REFERENCES [Credentials] ([CredentialId]) ON DELETE CASCADE
);


CREATE TABLE [WebCredentialAccesses] (
    [WebCredentialAccessId] uniqueidentifier NOT NULL,
    [WebCredentialId]       uniqueidentifier NOT NULL,
    [UserId]                uniqueidentifier NOT NULL,
    [PermissionLevel]       nvarchar(50)     NOT NULL,
    [ExpireAt]              datetime2        NOT NULL,
    [SharedAt]              datetime2        NOT NULL,
    [SharedByUserId]        uniqueidentifier NOT NULL,
    CONSTRAINT [PK_WebCredentialAccesses] PRIMARY KEY ([WebCredentialAccessId]),
    CONSTRAINT [FK_WebCredentialAccesses_AspNetUsers_SharedByUserId]
        FOREIGN KEY ([SharedByUserId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_WebCredentialAccesses_AspNetUsers_UserId]
        FOREIGN KEY ([UserId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_WebCredentialAccesses_WebCredentials_WebCredentialId]
        FOREIGN KEY ([WebCredentialId]) REFERENCES [WebCredentials] ([WebCredentialId]) ON DELETE CASCADE
);


CREATE TABLE [WebCredentialsHistorys] (
    [WebCredentialHistoryId] uniqueidentifier NOT NULL,
    [WebCredentialId]        uniqueidentifier NOT NULL,
    [Url]                    nvarchar(200)    NOT NULL,
    [Username]               nvarchar(100)    NOT NULL,
    [PasswordHash]           nvarchar(512)    NOT NULL,
    [Notes]                  nvarchar(100)    NULL,
    [ChangedByUserId]        uniqueidentifier NULL,
    [ChangedAt]              datetime2        NULL,
    [ChangeType]             nvarchar(50)     NOT NULL,
    CONSTRAINT [PK_WebCredentialsHistorys] PRIMARY KEY ([WebCredentialHistoryId]),
    CONSTRAINT [FK_WebCredentialsHistorys_AspNetUsers_ChangedByUserId]
        FOREIGN KEY ([ChangedByUserId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_WebCredentialsHistorys_WebCredentials_WebCredentialId]
        FOREIGN KEY ([WebCredentialId]) REFERENCES [WebCredentials] ([WebCredentialId]) ON DELETE CASCADE
);


CREATE TABLE [CreditCards] (
    [CreditDebitId]  uniqueidentifier NOT NULL,
    [CredentialId]   uniqueidentifier NOT NULL,
    [CardHolderName] nvarchar(100)    NOT NULL,
    [CardNumberHash] nvarchar(512)    NOT NULL,
    [ExpiryMonth]    nvarchar(2)      NOT NULL,
    [ExpiryYear]     nvarchar(2)      NOT NULL,
    [CvvHash]        nvarchar(512)    NOT NULL,
    [PinHash]        nvarchar(512)    NOT NULL,
    [BillingAddress] nvarchar(100)    NULL,
    [Notes]          nvarchar(200)    NULL,
    [CreatedAt]      datetime2        NOT NULL,
    [CreatorId]      uniqueidentifier NOT NULL,
    [EditedAt]       datetime2        NULL,
    [EditorId]       uniqueidentifier NULL,
    CONSTRAINT [PK_CreditCards] PRIMARY KEY ([CreditDebitId]),
    CONSTRAINT [FK_CreditCards_AspNetUsers_CreatorId]
        FOREIGN KEY ([CreatorId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_CreditCards_AspNetUsers_EditorId]
        FOREIGN KEY ([EditorId]) REFERENCES [AspNetUsers] ([Id]),
    CONSTRAINT [FK_CreditCards_Credentials_CredentialId]
        FOREIGN KEY ([CredentialId]) REFERENCES [Credentials] ([CredentialId]) ON DELETE CASCADE
);

CREATE TABLE [CreditCardsHistorys] (
    [CreditDebitHistoryId] uniqueidentifier NOT NULL,
